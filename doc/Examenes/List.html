def obtenerValor(numero)
  valor = ""
  i = 0
  nodo = @cabeza
  while i < numero do
    i = i + 1
    valor = nodo.value
    nodo = nodo.next
  end
  return valor
end

def push(*valor)
  for i in 0...valor.length
    nuevoNode = Node.new(valor[i], @nil, nil)
    if @numeroNodos == 0 then
      @cabeza = nuevoNode
      @cola = nuevoNode
    else
      @cabeza.previous = nuevoNode
      nuevoNode.next = @cabeza
      @cabeza = nuevoNode       
    end
    @numeroNodos = @numeroNodos + 1
  end
end

def sacarValor
  @valor = @cabeza.value
  if @numeroNodos == 1 then
    @cabeza = nil
    @cola = nil
  else
    @cabeza = @cabeza.next
  end
  @numeroNodos = @numeroNodos - 1
  return @valor
end
